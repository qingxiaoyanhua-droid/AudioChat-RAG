### Q2: 行动项如何提取？⭐⭐⭐⭐

**回答**：

**方法：Prompt + 规则提取**

```python
def extract_action_items(utterances: list) -> list:
    """
    行动项提取
    """
    # 1. 规则匹配（信号词）
    action_signals = [
        "我负责", "我来做", "下周完成", "预计完成",
        "action", "todo", "follow up"
    ]
    
    actions = []
    for u in utterances:
        for signal in action_signals:
            if signal in u.text:
                # 2. 提取关键信息
                action = {
                    "owner": u.speaker,
                    "task": extract_task(u.text),
                    "deadline": extract_deadline(u.text)
                }
                actions.append(action)
    
    # 3. LLM 校验（去除误报）
    validated_actions = llm_validate(actions)
    
    return validated_actions
```

### Q4: 说话人总结怎么做？⭐⭐⭐

**回答**：

**方法：按说话人分组 + LLM 总结**

```python
def generate_speaker_summary(utterances: list) -> dict:
    """
    说话人总结
    """
    # 1. 按说话人分组
    speaker_texts = {}
    for u in utterances:
        if u.speaker not in speaker_texts:
            speaker_texts[u.speaker] = []
        speaker_texts[u.speaker].append(u.text)
    
    # 2. LLM 总结每个说话人的发言
    summaries = {}
    for speaker, texts in speaker_texts.items():
        combined = "\n".join(texts)
        prompt = f"请总结以下发言内容：\n{combined}"
        summary = llm.generate(prompt)
        summaries[speaker] = summary
    
    return summaries
```

**效果**：
- 支持"张三说了什么"查询

## 📋 会议转录系统核心代码（完整版）

```python
class MeetingAnalysisSystem:
    """会议分析系统"""
    
    def __init__(self):
        self.asr = FunASRTranscriber()
        self.diarizer = ThreeDSpeakerDiarizer()
        self.rag = AudioChatRetriever()
        self.llm = FunAudioChatLLM()
    
    def analyze(self, audio_path: str) -> dict:
        """
        会议分析全流程
        """
        # 1. 语音转文字
        utterances = self.transcribe(audio_path)
        
        # 2. 说话人总结
        speaker_summaries = self.summarize_by_speaker(utterances)
        
        # 3. 会议纪要总结
        meeting_summary = self.summarize_meeting(utterances)
        
        # 4. 行动项提取
        action_items = self.extract_action_items(utterances)
        
        # 5. 决策项提取
        decisions = self.extract_decisions(utterances)
        
        # 6. 存储到 RAG
        self.rag.add_meeting_record(
            meeting_id=extract_id(audio_path),
            utterances=utterances,
            summary=meeting_summary,
            action_items=action_items,
            decisions=decisions
        )
        
        return {
            "utterances": utterances,
            "speaker_summaries": speaker_summaries,
            "meeting_summary": meeting_summary,
            "action_items": action_items,
            "decisions": decisions
        }
    
    def transcribe(self, audio_path: str) -> list:
        """语音转文字"""
        # 说话人分离
        diar_segments = self.diarizer.diarize(audio_path)
        
        # ASR 识别
        utterances = []
        for seg in diar_segments:
            text = self.asr.transcribe(seg.audio)
            utterances.append({
                "speaker": f"spk{seg.speaker}",
                "text": text,
                "start": seg.start,
                "end": seg.end
            })
        
        return utterances
    
    def summarize_by_speaker(self, utterances: list) -> dict:
        """说话人总结"""
        # 按说话人分组
        speaker_texts = {}
        for u in utterances:
            if u["speaker"] not in speaker_texts:
                speaker_texts[u["speaker"]] = []
            speaker_texts[u["speaker"]].append(u["text"])
        
        # LLM 总结
        summaries = {}
        for speaker, texts in speaker_texts.items():
            combined = "\n".join(texts)
            prompt = f"请总结以下发言内容：\n{combined}"
            summary = self.llm.generate(prompt)
            summaries[speaker] = summary
        
        return summaries
    
    def summarize_meeting(self, utterances: list) -> str:
        """会议纪要总结"""
        meeting_text = format_utterances(utterances)
        prompt = f"""请生成会议纪要，包含：
        1. 讨论要点
        2. 争议点
        3. 结论
        4. 行动项
        5. 决策项
        
        会议内容：
        {meeting_text}
        """
        return self.llm.generate(prompt)
    
    def extract_action_items(self, utterances: list) -> list:
        """行动项提取"""
        action_signals = ["我负责", "我来做", "下周完成", "预计完成", "action", "todo"]
        
        actions = []
        for u in utterances:
            for signal in action_signals:
                if signal in u["text"]:
                    action = {
                        "owner": u["speaker"],
                        "task": self.extract_task(u["text"]),
                        "deadline": self.extract_deadline(u["text"])
                    }
                    actions.append(action)
        
        # LLM 校验
        validated = self.llm_validate_actions(actions)
        return validated
    
    def extract_decisions(self, utterances: list) -> list:
        """决策项提取"""
        decision_signals = ["确定", "决定", "通过", "同意", "confirm", "decide"]
        
        decisions = []
        for u in utterances:
            for signal in decision_signals:
                if signal in u["text"]:
                    decision = {
                        "content": self.extract_decision_content(u["text"]),
                        "time": self.extract_time(u["text"]),
                        "participants": [u["speaker"]]
                    }
                    decisions.append(decision)
        
        return decisions
```
